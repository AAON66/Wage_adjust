class ApiKeyService:
    def __init__(self, db: Session):
        self.db = db

    def create_key(self, *, name: str, rate_limit: int = 1000, expires_at: datetime | None, created_by: str) -> tuple[ApiKey, str]:
        """创建新 Key，返回 (ORM model, 明文 key)。明文仅此一次可获取。"""
        # 1. 生成随机 Key: secrets.token_urlsafe(32) → 43 char base64url
        # 2. SHA-256 hash: hashlib.sha256(plain_key.encode()).hexdigest()
        # 3. key_prefix: plain_key[:8]
        # 4. 存入 DB, commit, 返回 (model, plain_key)

    def validate_key(self, plain_key: str, *, client_ip: str | None = None) -> ApiKey | None:
        """验证 Key。成功时更新 last_used_at/last_used_ip。返回 None 表示无效/撤销/过期。"""
        # 1. SHA-256 hash plain_key
        # 2. SELECT * FROM api_keys WHERE key_hash = ? AND is_active = True
        # 3. 检查 expires_at: if expires_at and expires_at < utc_now() → return None
        # 4. 更新 last_used_at = utc_now(), last_used_ip = client_ip
        # 5. commit, return api_key

    def rotate_key(self, key_id: str, *, created_by: str) -> tuple[ApiKey, str]:
        """轮换：撤销旧 Key，创建同名同配置新 Key。"""
        # 1. 查找 key_id, 标记 is_active = False
        # 2. 创建新 Key (同 name, rate_limit, expires_at)
        # 3. 返回 (new_key, new_plain_key)

    def revoke_key(self, key_id: str) -> ApiKey:
        """撤销 Key: 设置 is_active = False"""

    def list_keys(self) -> list[ApiKey]:
        """列出所有 Key（含已撤销）"""

    def get_key(self, key_id: str) -> ApiKey | None:
        """查询单个 Key"""

**2. 创建 `backend/app/utils/cursor_pagination.py`** — 游标编解码工具（per D-05, D-06）：

```python
import base64
import json
from sqlalchemy.orm import Query


def encode_cursor(last_id: str, sort_value: str | None = None) -> str:
    """Base64 编码游标。per D-05: base64(json({"id": last_id, "sort": sort_value}))"""
    payload = json.dumps({'id': last_id, 'sort': sort_value}, separators=(',', ':'))
    return base64.urlsafe_b64encode(payload.encode()).decode()


def decode_cursor(cursor: str) -> dict[str, str | None]:
    """解码游标。返回 {"id": ..., "sort": ...}。无效游标抛 ValueError。"""
    try:
        payload = base64.urlsafe_b64decode(cursor.encode()).decode()
        data = json.loads(payload)
        if 'id' not in data:
            raise ValueError('Missing id in cursor')
        return data
    except Exception as exc:
        raise ValueError(f'Invalid cursor: {exc}') from exc


def apply_cursor_pagination(
    query,  # SQLAlchemy select statement
    *,
    cursor: str | None = None,
    page_size: int = 20,
    id_column,  # SQLAlchemy column for ID
    sort_column=None,  # Optional sort column
):
    """应用游标分页到 SQLAlchemy select。返回 (paginated_query, page_size)。
    per D-06: page_size 截断为 max 100, 默认 20。"""
    page_size = min(max(page_size, 1), 100)

    if cursor:
        data = decode_cursor(cursor)
        last_id = data['id']
        # 使用 id_column > last_id 确保不重复
        query = query.where(id_column > last_id)

    # 按 id 排序确保稳定顺序
    query = query.order_by(id_column.asc())
    # 多取一条用于判断 has_more
    query = query.limit(page_size + 1)
    return query, page_size
```

    self,
    *,
    cycle_id: str | None = None,
    department: str | None = None,
    cursor: str | None = None,
    page_size: int = 20,
) -> tuple[list[EmployeeSubmission], str | None, bool]:
    """游标分页查询已审批调薪结果（per D-07, D-08）。
    返回 (items, next_cursor, has_more)"""
    from backend.app.utils.cursor_pagination import apply_cursor_pagination, encode_cursor

    query = (
        self._submission_query()
        .join(EmployeeSubmission.ai_evaluation)
        .join(AIEvaluation.salary_recommendation)
        .where(SalaryRecommendation.status == 'approved')
    )
    if cycle_id:
        query = query.where(EmployeeSubmission.cycle_id == cycle_id)
    if department:
        query = query.join(EmployeeSubmission.employee).where(Employee.department == department)

    query, page_size = apply_cursor_pagination(
        query, cursor=cursor, page_size=page_size,
        id_column=EmployeeSubmission.id,
    )
    results = list(self.db.scalars(query))
    has_more = len(results) > page_size
