       ```typescript
       export async function computeFileSHA256(file: File): Promise<string> {
         const buffer = await file.arrayBuffer();
         const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
         const hashArray = Array.from(new Uint8Array(hashBuffer));
         return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
       }
       ```

       **handleFilesSelected(files: FileList | null):**
       ```
       1. Convert FileList to array
       2. Create queue: files.map(f => ({ file: f, status: 'pending' as const }))
       ```

       **processQueue(queue: FileQueueItem[]):**
       ```
       For each item in queue where status === 'pending':
         a. Update item status to 'checking', setHashCheckStatus('checking')
         b. try:
              hash = await computeFileSHA256(item.file)
              result = await checkDuplicate(hash, submissionId)  // submissionId from page context
              setHashCheckStatus('idle')
            catch:
              setHashCheckStatus('error')
              // Graceful degradation: mark as clean, allow upload without check
              Update item status to 'clean'
              continue to next item
         c. If result.is_duplicate:
              Update item with duplicateInfo and status = 'currentDuplicate'
              setFileQueue([...queue])  // trigger re-render, show modal
              RETURN — wait for user decision via modal callbacks
         d. If not duplicate:
              Update item status to 'clean'
              continue to next item

       After all items checked:
         Upload all 'clean' items via normal uploadSubmissionFiles()
         Upload all 'approvedToUpload' items individually via uploadSubmissionFilesWithDuplicate()
         Mark all uploaded as 'completed'
         Trigger parse as before
       ```

       **handleDuplicateConfirm():**

       ```
       Find item with status === 'currentDuplicate'
       Update its status to 'approvedToUpload'
       setFileQueue([...queue])
       Resume processQueue with remaining pending items
       ```

       ```
       Find item with status === 'currentDuplicate'
       Update its status to 'skipped'
       setFileQueue([...queue])
       Resume processQueue with remaining pending items
       ```
