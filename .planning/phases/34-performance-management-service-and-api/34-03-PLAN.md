     class PerformanceService:
         def __init__(
             self,
             db: Session,
             *,
             settings: Settings | None = None,
             cache: TierCache | None = None,
             engine: PerformanceTierEngine | None = None,

   - 6 个公开方法：
     - `list_records(*, year=None, department=None, page=1, page_size=50) -> tuple[list[PerformanceRecord], int]`：joinedload(Employee)；按 year + department filter；按 created_at desc 排序；返回 (items, total)
     - `create_record(*, employee_id: str, year: int, grade: str, source: str = 'manual') -> PerformanceRecord`：
       1. grade.strip().upper() → A/B/C/D/E 校验，否则 raise ValueError
       2. 查 employee（不存在 → ValueError）
       3. department_snapshot = employee.department（D-08：可为 None）
       4. UPSERT on (employee_id, year)；source 用入参值
       5. db.commit() + db.refresh + return
     - `list_available_years() -> list[int]`：`SELECT DISTINCT year FROM performance_records ORDER BY year DESC`；空时返回 `[date.today().year]`
     - `invalidate_tier_cache(years: int | Iterable[int]) -> None`：转 set + cache.invalidate(year) per year
     - `get_tier_summary(year: int) -> TierSummaryResponse | None`：
       1. cache.get_cached(year) hit → 反序列化 datetime → 返回
       2. miss → 查 PerformanceTierSnapshot by year
       3. 表也 miss → 返回 None（API 层转 404）
       4. 表 hit → 转 TierSummaryResponse + cache.set_cached(year, dict) → 返回
       - tiers_count 计算：从 tiers_json 统计 1/2/3/None 数量
     - `recompute_tiers(year: int) -> TierSummaryResponse`：
       1. **Step 1**：确保 snapshot 行存在 — 用 `INSERT ... ON CONFLICT (year) DO NOTHING`（SQLite 用 `INSERT OR IGNORE`，PostgreSQL 用 `ON CONFLICT DO NOTHING`；用 SQLAlchemy `dialect_kwargs` 适配；最简方式：try insert + catch IntegrityError）
       2. **Step 2**：行锁 — `SELECT id FROM performance_tier_snapshots WHERE year=:year FOR UPDATE NOWAIT`（SQLite 不支持 FOR UPDATE，dev 环境降级为普通 SELECT；用 `text()` + `dialect.name` 检测；如 dialect 是 sqlite，warn log + 跳过 NOWAIT；prod PostgreSQL 路径必须真实加锁）。NOWAIT 失败（OperationalError 含 "could not obtain lock" / "lock not available" / "could not serialize"）→ 抛 `TierRecomputeBusyError(year)`
       3. **Step 3**：拉绩效数据 — `select(PerformanceRecord.employee_id, PerformanceRecord.grade).where(year=year)` → list[tuple]
       4. **Step 4**：调 Engine — `result = self.engine.assign(rows)`
       5. **Step 5**：写回 snapshot — UPDATE tiers_json / sample_size / insufficient_sample / distribution_warning / actual_distribution_json / skipped_invalid_grades；updated_at 自动更新
       6. **Step 6**：commit → refresh
       7. **Step 7**：构建 TierSummaryResponse + cache.set_cached
       8. **Step 8**：return
       - 任何 step 3-6 内异常 catch → rollback → raise TierRecomputeFailedError(year, str(exc)) from exc
       - actual_distribution dict[int, float] → 转 dict[str, float]（JSON 兼容）
       - tiers_count: 从 result.tiers 用 collections.Counter 统计

   - 私有 helper：`_snapshot_to_summary(snapshot) -> TierSummaryResponse`（公共反序列化逻辑，复用于 get + recompute）

   - 关键实现细节：
     - **行锁 SQLite 兼容**：用 `if self.db.bind.dialect.name == 'sqlite': skip NOWAIT lock with warning log` 否则 `db.execute(text('SELECT id FROM performance_tier_snapshots WHERE year=:year FOR UPDATE NOWAIT'), {'year': year})`；OperationalError 含 "could not obtain lock" / "lock not available" → TierRecomputeBusyError

     tier_recompute_status: str | None = None  # W-1
     tier_recompute_completed_at = None
     if (
         job.import_type == 'performance_grades'
         and job.status in ('completed', 'partial')
     ):

         from backend.app.core.config import get_settings as _get_settings
         from backend.app.core.redis import get_redis as _get_redis

             TierRecomputeBusyError,

         # 提取受影响 years：从 result_summary['execution']['rows'] 抽 year
         exec_rows = (job.result_summary or {}).get('execution', {}).get('rows', [])
         affected_years: set[int] = set()
         # 兜底：如果 rows 不含 year，查 source='excel' 最近 1 分钟内的不同 year
         if not affected_years:
             from datetime import datetime as _dt, timedelta, timezone as _tz
             one_minute_ago = _dt.now(_tz.utc) - timedelta(minutes=1)
             year_rows = self.db.execute(
                 select(PerformanceRecord.year).where(
                     PerformanceRecord.source == 'excel',
                     PerformanceRecord.updated_at >= one_minute_ago,
                 ).distinct()
             ).scalars().all()
             affected_years = set(year_rows)

         settings = _get_settings()
         try:
             redis_client = _get_redis()
         except Exception:
             redis_client = None
         cache = TierCache(redis_client=redis_client, settings=settings)
         perf_service = PerformanceService(self.db, settings=settings, cache=cache)

         # invalidate cache for all affected years
         perf_service.invalidate_tier_cache(affected_years)

         # 同步重算（最长 5 秒），D-03
         timeout_s = settings.performance_tier_recompute_timeout_seconds
         tier_recompute_status = 'completed'
         busy_seen = False
         failed_seen = False
         try:
             with ThreadPoolExecutor(max_workers=1) as ex:
                 def _do() -> None:
                     nonlocal busy_seen, failed_seen
                     for y in sorted(affected_years):
                         try:
                             perf_service.recompute_tiers(y)
                         except TierRecomputeBusyError:
                             busy_seen = True  # D-06：自动撞手动 → busy_skipped
                         except TierRecomputeFailedError as fe:
                             failed_seen = True
                             logger.warning('tier_recompute_error year=%s: %s', y, fe.cause)
                 fut = ex.submit(_do)
                 try:
                     fut.result(timeout=timeout_s)
                     tier_recompute_completed_at = datetime.now(timezone.utc)
                     if failed_seen:
                         tier_recompute_status = 'failed'
                     elif busy_seen:
                         tier_recompute_status = 'busy_skipped'
                     else:
                         tier_recompute_status = 'completed'
                 except FutureTimeout:
                     tier_recompute_status = 'in_progress'
                     # fut 在后台继续运行，executor 不被强制 shutdown
         except Exception as exc:
             logger.exception('Tier recompute hook failed: %s', exc)
             tier_recompute_status = 'failed'
