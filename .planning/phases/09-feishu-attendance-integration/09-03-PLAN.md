API 端点:
- GET  /api/v1/feishu/config -> FeishuConfigRead
- POST /api/v1/feishu/config -> FeishuConfigRead (body: FeishuConfigCreate)
- PUT  /api/v1/feishu/config/{id} -> FeishuConfigRead (body: FeishuConfigUpdate)
- POST /api/v1/feishu/test-connection -> { success: bool, message: str }
- POST /api/v1/feishu/sync -> SyncResultResponse (body: { mode: 'full' | 'incremental' })
- GET  /api/v1/feishu/sync-status -> SyncStatusRead
- GET  /api/v1/attendance/{employee_id} -> AttendanceSummary
- GET  /api/v1/attendance/?department_id=&search=&offset=&limit= -> { items: AttendanceRecordRead[], total: int }

```typescript
interface FieldMapping {
  employee_no: string;
  attendance_rate: string;
  absence_days: string;
  overtime_hours: string;
  late_count: string;
  early_leave_count: string;
}

interface FeishuConfigRead {
  id: string;
  app_id: string;
  app_secret_masked: string;
  bitable_app_token: string;
  bitable_table_id: string;
  sync_hour: number;
  sync_minute: number;
  field_mapping: FieldMapping;
  is_active: boolean;
}

interface AttendanceSummary {
  employee_id: string;
  employee_no: string;
  attendance_rate: number | null;
  absence_days: number | null;
  overtime_hours: number | null;
  late_count: number | null;
  early_leave_count: number | null;
  synced_at: string | null;
  is_stale: boolean;
}

interface SyncStatusRead {
  id: string;
  mode: string;
  status: 'success' | 'failed' | 'running';
  total_records: number;
  matched_records: number;
  unmatched_records: number;
  error_message: string | null;
  started_at: string;
  finished_at: string | null;
}
```
