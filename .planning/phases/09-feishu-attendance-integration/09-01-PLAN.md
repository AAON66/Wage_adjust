```python
class FeishuConfig(UUIDPrimaryKeyMixin, CreatedAtMixin, UpdatedAtMixin, Base):
    __tablename__ = 'feishu_configs'

    app_id: Mapped[str] = mapped_column(String(128), nullable=False)
    encrypted_app_secret: Mapped[str] = mapped_column(String(512), nullable=False)
    bitable_app_token: Mapped[str] = mapped_column(String(128), nullable=False)
    bitable_table_id: Mapped[str] = mapped_column(String(128), nullable=False)
    field_mapping: Mapped[str] = mapped_column(Text, nullable=False, default='{}')  # JSON: {"飞书字段名": "系统字段名"}
    sync_hour: Mapped[int] = mapped_column(Integer, nullable=False, default=6)
    sync_minute: Mapped[int] = mapped_column(Integer, nullable=False, default=0)
    sync_timezone: Mapped[str] = mapped_column(String(64), nullable=False, default='Asia/Shanghai')  # 【Review #2 fix】显式时区
    is_active: Mapped[bool] = mapped_column(Boolean, nullable=False, default=True)
```

加密/解密 app_secret 的 property 方法：使用独立的 `FEISHU_ENCRYPTION_KEY` 环境变量，复用 `encrypt_national_id` / `decrypt_national_id` 函数，传入不同的 key bytes。

**2. backend/app/models/attendance_record.py** — 考勤记录模型（per D-11, ATT-06）

```python
class AttendanceRecord(UUIDPrimaryKeyMixin, CreatedAtMixin, UpdatedAtMixin, Base):
    __tablename__ = 'attendance_records'
    __table_args__ = (
        UniqueConstraint('employee_id', 'period', name='uq_attendance_employee_period'),
        UniqueConstraint('feishu_record_id', name='uq_attendance_feishu_record_id'),
    )

    employee_id: Mapped[str] = mapped_column(String(36), ForeignKey('employees.id'), nullable=False)
    employee_no: Mapped[str] = mapped_column(String(64), nullable=False, index=True)
    period: Mapped[str] = mapped_column(String(32), nullable=False)  # 如 '2026-03'
    attendance_rate: Mapped[float | None] = mapped_column(Float, nullable=True)
    absence_days: Mapped[float | None] = mapped_column(Float, nullable=True)
    overtime_hours: Mapped[float | None] = mapped_column(Float, nullable=True)
    late_count: Mapped[int | None] = mapped_column(Integer, nullable=True)
    early_leave_count: Mapped[int | None] = mapped_column(Integer, nullable=True)
    feishu_record_id: Mapped[str | None] = mapped_column(String(128), nullable=True)
    source_modified_at: Mapped[int | None] = mapped_column(BigInteger, nullable=True)  # 飞书 last_modified_time 毫秒戳
    data_as_of: Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)  # 【Review #5 fix】面向业务的"数据截至"时间戳，用于 ATT-06 前端展示
    synced_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), nullable=False)  # 同步完成时间
```

**Review concern #5 addressed:** `data_as_of` 是面向用户展示的"数据截至"时间戳。取值规则：若飞书记录有 `source_modified_at`，则 `data_as_of = datetime.fromtimestamp(source_modified_at / 1000, tz=timezone.utc)`；否则 `data_as_of = synced_at`。

```python
class FieldMappingItem(BaseModel):
    feishu_field: str          # 飞书多维表格字段名
    system_field: str          # 系统字段名（attendance_rate 等）

class FeishuConfigCreate(BaseModel):
    app_id: str
    app_secret: str            # 明文传入，后端加密存储
    bitable_app_token: str
    bitable_table_id: str
    field_mapping: list[FieldMappingItem]
    sync_hour: int = 6
    sync_minute: int = 0
    sync_timezone: str = 'Asia/Shanghai'

class FeishuConfigUpdate(BaseModel):
    app_id: str | None = None
    app_secret: str | None = None   # 【Review #10 fix】None/空字符串表示保留原值，非空则更新
    bitable_app_token: str | None = None
    bitable_table_id: str | None = None
    field_mapping: list[FieldMappingItem] | None = None
    sync_hour: int | None = None
    sync_minute: int | None = None
    sync_timezone: str | None = None

class FeishuConfigRead(BaseModel):
    id: str
    app_id: str
    app_secret_masked: str     # 仅返回最后 4 位如 "****xxxx"
    bitable_app_token: str
    bitable_table_id: str
    field_mapping: list[FieldMappingItem]
    sync_hour: int
    sync_minute: int
    sync_timezone: str
    is_active: bool
    model_config = ConfigDict(from_attributes=True)

class SyncTriggerRequest(BaseModel):
    mode: str  # 'full' | 'incremental'

class SyncTriggerResponse(BaseModel):
    sync_log_id: str
    status: str
    message: str

class SyncLogRead(BaseModel):
    id: str
    mode: str
    status: str
    total_fetched: int
    synced_count: int
    updated_count: int
    skipped_count: int
    unmatched_count: int
    failed_count: int
    error_message: str | None
    unmatched_employee_nos: list[str] | None  # parsed from JSON
    started_at: datetime
    finished_at: datetime | None
    triggered_by: str | None
    model_config = ConfigDict(from_attributes=True)

class FeishuConfigExistsResponse(BaseModel):
    exists: bool
```
