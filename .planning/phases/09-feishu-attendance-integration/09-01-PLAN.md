   ```python
   class FieldMapping(BaseModel):
       employee_no: str          # 飞书中的工号字段名（关联键，必填）
       attendance_rate: str = ''
       absence_days: str = ''
       overtime_hours: str = ''
       late_count: str = ''
       early_leave_count: str = ''

   class FeishuConfigCreate(BaseModel):
       app_id: str
       app_secret: str           # 明文传入，后端加密存储
       bitable_app_token: str
       bitable_table_id: str
       sync_hour: int = 6        # 0-23
       sync_minute: int = 0      # 0-59
       field_mapping: FieldMapping

   class FeishuConfigRead(BaseModel):
       model_config = ConfigDict(from_attributes=True)
       id: str
       app_id: str
       app_secret_masked: str    # 仅最后 4 位，如 '****cdef'
       bitable_app_token: str
       bitable_table_id: str
       sync_hour: int
       sync_minute: int
       field_mapping: FieldMapping
       is_active: bool

   class FeishuConfigUpdate(BaseModel):
       app_id: str | None = None
       app_secret: str | None = None
       bitable_app_token: str | None = None
       bitable_table_id: str | None = None
       sync_hour: int | None = None
       sync_minute: int | None = None
       field_mapping: FieldMapping | None = None

   class SyncTriggerRequest(BaseModel):
       mode: str = 'incremental'  # 'full' | 'incremental'

   class SyncResultResponse(BaseModel):
       status: str                # 'success' | 'failed' | 'running'
       total_records: int = 0
       matched_records: int = 0
       unmatched_records: int = 0
       error_message: str | None = None

   class SyncStatusRead(BaseModel):
       model_config = ConfigDict(from_attributes=True)
       id: str
       mode: str
       status: str
       total_records: int
       matched_records: int
       unmatched_records: int
       error_message: str | None
       started_at: datetime
   ```
