2. 创建 `backend/app/scheduler/feishu_scheduler.py`（per D-05, D-06, ATT-03）:
   ```python
   from apscheduler.schedulers.asyncio import AsyncIOScheduler
   from apscheduler.triggers.cron import CronTrigger
   import logging
   from backend.app.core.database import SessionLocal

   logger = logging.getLogger(__name__)
   scheduler = AsyncIOScheduler()

   def _run_incremental_sync():
       """定时同步 job — 在独立 DB Session 中执行（per 陷阱 1）"""
       db = SessionLocal()
       try:
           from backend.app.services.feishu_service import FeishuService
           service = FeishuService(db)
           result = service.sync_with_retry(mode='incremental')
           db.commit()
           logger.info('定时考勤同步完成: %s', result)
       except Exception:
           db.rollback()
           logger.exception('定时考勤同步失败')
       finally:
           db.close()

   def start_scheduler(sync_hour: int = 6, sync_minute: int = 0):
       # 从数据库读取配置的同步时间
       db = SessionLocal()
       try:
           from backend.app.services.feishu_service import FeishuService
           config = FeishuService(db).get_config()
           if config:
               sync_hour = config.sync_hour
               sync_minute = config.sync_minute
       finally:
           db.close()

       scheduler.add_job(
           _run_incremental_sync,
           trigger=CronTrigger(hour=sync_hour, minute=sync_minute),
           id='feishu_attendance_sync',
           replace_existing=True,
       )
       scheduler.start()
       logger.info('考勤同步调度器已启动: 每天 %02d:%02d', sync_hour, sync_minute)

   def stop_scheduler():
       if scheduler.running:
           scheduler.shutdown(wait=False)

   def reschedule(sync_hour: int, sync_minute: int):
       """配置更新后重新调度"""
       scheduler.reschedule_job(
           'feishu_attendance_sync',
           trigger=CronTrigger(hour=sync_hour, minute=sync_minute),
       )
       logger.info('考勤同步已重新调度: 每天 %02d:%02d', sync_hour, sync_minute)
   ```
