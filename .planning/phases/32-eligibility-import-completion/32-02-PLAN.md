```python
def _parse_excel_date(value) -> date | None:
    if value is None or (isinstance(value, str) and not value.strip()):
        return None
    if isinstance(value, (int, float)):
        return pd.to_datetime(value, unit='D', origin='1899-12-30').date()
    s = str(value).strip()
    if s.isdigit() and 5 <= len(s) <= 6:  # Excel serial range

    return pd.to_datetime(s).date()
```

    ```python
    @staticmethod
    def _parse_excel_date(value) -> 'date | None':
        """解析 Excel 单元格日期值。Pitfall 1: dtype=str 读出的 Excel 序列号需双分支处理。"""
        from datetime import date as _date
        import pandas as _pd
        if value is None:
            return None
        if isinstance(value, _date):
            return value
        if isinstance(value, (int, float)):
            return _pd.to_datetime(value, unit='D', origin='1899-12-30').date()
        s = str(value).strip()
        if not s:
            return None
        if s.isdigit() and 5 <= len(s) <= 6:
            return _pd.to_datetime(int(s), unit='D', origin='1899-12-30').date()
        try:
            return _pd.to_datetime(s).date()
        except Exception as exc:
            raise ValueError(f'入职日期格式无效: {value!r} ({exc})') from exc
    ```

    ```python
    def _import_hire_info(
        self,
        dataframe: pd.DataFrame,
        *,
        overwrite_mode: str = 'merge',
    ) -> list[dict[str, object]]:
        """D-02 + IMPORT-05: 导入入职信息到 Employee 表，业务键 = employee_id。"""
        results: list[dict[str, object]] = []
        # employee_no → Employee 映射（用 IdentityBindingService 一次 prefetch）
        employee_nos = [str(v).strip() for v in dataframe.get('employee_no', []) if v is not None]
        binding = IdentityBindingService(self.db)
        emp_map = binding.find_employees_by_nos(employee_nos)  # 若无此方法，参考 _import_certifications 实现自查找

        for index, row in dataframe.iterrows():
            row_no = int(index) + 1
            emp_no = str(row.get('employee_no', '')).strip()
            if not emp_no:
                results.append({
                    'row': row_no, 'status': 'failed', 'employee_no': '',
                    'error': self._localize_error_message('employee_no is required'),
                })
                continue
            employee = emp_map.get(emp_no)
            if employee is None:
                results.append({
                    'row': row_no, 'status': 'failed', 'employee_no': emp_no,
                    'error': '未找到员工工号',
                })
                continue

            raw_hire = row.get('hire_date')
            raw_last_adj = row.get('last_salary_adjustment_date')
            try:
                new_hire = self._parse_excel_date(raw_hire)
                new_last_adj = self._parse_excel_date(raw_last_adj)
            except ValueError as exc:
                results.append({
                    'row': row_no, 'status': 'failed', 'employee_no': emp_no,
                    'error': self._localize_error_message(str(exc)),
                })
                continue

            # D-12 必填字段空值在 replace 模式下 → failed
            if new_hire is None and overwrite_mode == 'replace':
                results.append({
                    'row': row_no, 'status': 'failed', 'employee_no': emp_no,

                })
                continue

            # merge: 空值保留旧值；replace: 时间戳字段也保留（CONTEXT D-子提示）
            old_hire = employee.hire_date
            old_last_adj = employee.last_salary_adjustment_date
            did_change = False

            if new_hire is not None and new_hire != old_hire:
                employee.hire_date = new_hire
                did_change = True

            if new_last_adj is not None and new_last_adj != old_last_adj:
                employee.last_salary_adjustment_date = new_last_adj
                did_change = True

            results.append({
                'row': row_no, 'status': 'success',
                'employee_no': emp_no,
                'action': 'update' if did_change else 'no_change',
                'fields': {
                    'hire_date': {'old': old_hire.isoformat() if old_hire else None,
                                  'new': new_hire.isoformat() if new_hire else None},
                    'last_salary_adjustment_date': {
                        'old': old_last_adj.isoformat() if old_last_adj else None,
                        'new': new_last_adj.isoformat() if new_last_adj else None,
                    },
                },
            })
        self.db.flush()
        return results
    ```
