      ```python
      def invalidate_for_event(self, event_type: str, cycle_id: str | None = None) -> int:
          """根据业务事件清除相关缓存。
          event_type 取值:
          - 'approval': 审批操作后，清除 approval_pipeline + kpi 相关 key
          - 'evaluation': 评估完成后，清除 ai_level + salary_dist + approval_pipeline 相关 key
          - 'import': 批量导入后，清除所有看板缓存
          - 'all': 清除所有看板缓存
          返回删除的 key 数量。
          """
          if event_type == 'all' or event_type == 'import':
              pattern = f'{self.KEY_PREFIX}:*'
          elif cycle_id:
              # 按周期清除特定类型
              chart_types = {
                  'approval': ['approval_pipeline'],
                  'evaluation': ['ai_level', 'salary_dist', 'approval_pipeline', 'dept_insight', 'dept_drilldown_*'],
              }.get(event_type, ['*'])
              deleted = 0
              for ct in chart_types:
                  pattern = f'{self.KEY_PREFIX}:{cycle_id}:*:{ct}'
                  keys = self.redis.keys(pattern)
                  if keys:
                      deleted += self.redis.delete(*keys)
              return deleted
          else:
              pattern = f'{self.KEY_PREFIX}:*'
          keys = self.redis.keys(pattern)
          if keys:
              return self.redis.delete(*keys)
          return 0
      ```
