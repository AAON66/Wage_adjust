     ```python
     def _load_contributors_for_approval(self, submission_id: str) -> list[ContributorRead]:
         owner_files = list(self.db.scalars(
             select(UploadedFile).where(UploadedFile.submission_id == submission_id)
         ))
         result: list[ContributorRead] = []
         for file in owner_files:
             contributors = list(self.db.scalars(
                 select(ProjectContributor).where(ProjectContributor.uploaded_file_id == file.id)
             ))
             if not contributors:
                 continue
             total_contributor_pct = sum(c.contribution_pct for c in contributors)
             owner_pct = 100.0 - total_contributor_pct
             owner_submission = self.db.get(EmployeeSubmission, submission_id)
             if owner_submission and owner_submission.employee:
                 result.append(ContributorRead(
                     employee_id=owner_submission.employee_id,
                     employee_name=owner_submission.employee.name,
                     contribution_pct=round(owner_pct, 2),
                     uploaded_file_id=file.id,
                     file_name=file.file_name,
                 ))
             for contributor in contributors:
                 contributor_submission = self.db.get(EmployeeSubmission, contributor.submission_id)
                 if contributor_submission and contributor_submission.employee:
                     result.append(ContributorRead(
                         employee_id=contributor_submission.employee_id,
                         employee_name=contributor_submission.employee.name,
                         contribution_pct=round(contributor.contribution_pct, 2),
                         uploaded_file_id=file.id,
                         file_name=file.file_name,
                     ))
         return result
     ```
