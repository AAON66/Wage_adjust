     ```python
     def _load_shared_evidence(self, submission_id: str) -> list[EvidenceItem]:
         contributor_rows = list(self.db.scalars(
             select(ProjectContributor).where(ProjectContributor.submission_id == submission_id)
         ))
         scaled_items: list[EvidenceItem] = []
         for contributor in contributor_rows:
             owner_file = self.db.get(UploadedFile, contributor.uploaded_file_id)
             if owner_file is None:
                 continue
             owner_evidence = list(self.db.scalars(
                 select(EvidenceItem).where(
                     EvidenceItem.submission_id == owner_file.submission_id,
                     EvidenceItem.metadata_json['file_id'].as_string() == owner_file.id,
                 )
             ))
             ratio = contributor.contribution_pct / 100.0
             for item in owner_evidence:
                 scaled = EvidenceItem(
                     submission_id=submission_id,
                     source_type=item.source_type,
                     title=item.title,
                     content=item.content,
                     confidence_score=round(item.confidence_score * ratio, 4),
                     metadata_json={
                         **item.metadata_json,
                         'contribution_pct': contributor.contribution_pct,
                         'shared_file_id': contributor.uploaded_file_id,
                     },
                 )
                 # Transient — do NOT add to session
                 scaled_items.append(scaled)
         return scaled_items
     ```
