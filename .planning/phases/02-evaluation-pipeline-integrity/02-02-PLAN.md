```python
class RedisRateLimiter:
    def __init__(self, redis_client, *, key: str, limit: int, window_seconds: int = 60) -> None: ...
    def acquire(self) -> None:
        # ZADD/ZREMRANGEBYSCORE sliding window
        pipe = self.redis.pipeline()
        pipe.zremrangebyscore(self.key, '-inf', now - self.window_seconds)
        pipe.zadd(self.key, {str(now): now})
        pipe.zcard(self.key)
        pipe.expire(self.key, self.window_seconds + 1)
        results = pipe.execute()
        count = results[2]
        if count > self.limit:
            self.redis.zrem(self.key, str(now))
            raise RuntimeError('DeepSeek rate limit reached for the current minute window.')
```
