       def extract_image_text(self, image_path: str | Path) -> DeepSeekCallResult:
           """Extract text from an image file using DeepSeek vision API."""
           from pathlib import Path as _Path
           import base64
           path = _Path(image_path)
           try:
               from PIL import Image as _Image
               img = _Image.open(path)
               # Resize if > 1MB source file
               if path.stat().st_size > 1_000_000:
                   img.thumbnail((1024, 1024), _Image.LANCZOS)
               import io
               buf = io.BytesIO()
               img_format = img.format or 'PNG'
               img.save(buf, format=img_format)
               image_bytes = buf.getvalue()
               mime_type = f'image/{img_format.lower()}'
           except Exception:
               image_bytes = path.read_bytes()
               mime_type = f'image/{path.suffix.lstrip(".").lower() or "png"}'
           image_b64 = base64.b64encode(image_bytes).decode('utf-8')
           messages = self.prompt_library.build_image_ocr_messages(image_b64, mime_type)
           return self._invoke_json('image_ocr', messages)
       ```

    3. Add _enrich_image_document method to ParseService:

       def _enrich_image_document(
           self,
           parsed: ParsedDocument,
           file_path: str | Path,
       ) -> ParsedDocument:
           """Call DeepSeek vision API to extract text from an image file.

           Returns parsed with enriched text when DeepSeek is configured and
           the image contains text. Returns parsed with ocr_skipped=True metadata
           when DeepSeek is unconfigured or the call fails.
           """
           if self.deepseek_service is None:
               return ParsedDocument(
                   text='',
                   metadata={**parsed.metadata, 'ocr_skipped': True, 'reason': 'deepseek_not_configured'},
               )
           try:
               result = self.deepseek_service.extract_image_text(file_path)
               if result.payload.get('has_text'):
                   return ParsedDocument(
                       text=result.payload.get('extracted_text', ''),
                       metadata=parsed.metadata,
                   )
               return ParsedDocument(
                   text='',
                   metadata={**parsed.metadata, 'ocr_skipped': True, 'reason': 'no_text_detected'},
               )
           except Exception as exc:
               logger.warning('Image OCR failed for %s: %s', file_path, exc)
               return ParsedDocument(
                   text='',
                   metadata={**parsed.metadata, 'ocr_skipped': True, 'reason': 'ocr_failed'},
               )
       ```
