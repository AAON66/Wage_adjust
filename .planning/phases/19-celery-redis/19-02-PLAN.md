   ```dockerfile
   WORKDIR /app

   COPY requirements.txt .
   RUN pip install --no-cache-dir -r requirements.txt

   COPY . .

   EXPOSE 8011

   CMD ["uvicorn", "backend.app.main:app", "--host", "0.0.0.0", "--port", "8011"]
   ```

2. 创建项目根目录下 `docker-compose.yml`（per D-09），包含三个服务：

   - **redis** 服务：
     - image: `redis:7-alpine`
     - ports: `"6379:6379"`
     - volumes: `redis_data:/data`
     - healthcheck: `redis-cli ping`，interval 10s，timeout 5s，retries 3

   - **backend** 服务：
     - build: `.`（使用上面的 Dockerfile）
     - command: `uvicorn backend.app.main:app --host 0.0.0.0 --port 8011 --reload`
     - ports: `"8011:8011"`
     - env_file: `.env`
     - environment: `REDIS_URL=redis://redis:6379/0`（覆盖 .env 中的 localhost 为 docker 网络中的 redis 服务名）
     - depends_on: redis（condition: service_healthy）
     - volumes: `.:/app`

   - **celery-worker** 服务（per D-09, D-10）：
     - build: `.`
     - command: `celery -A backend.app.celery_app worker --loglevel=info --concurrency=2`（per D-01 启动命令，per D-10 concurrency=2）
     - env_file: `.env`
     - environment: `REDIS_URL=redis://redis:6379/0`
     - depends_on: redis（condition: service_healthy）
     - volumes: `.:/app`

   - volumes 声明：`redis_data:`

注意：不使用 `version:` 字段（Docker Compose V2 已弃用此字段）。
